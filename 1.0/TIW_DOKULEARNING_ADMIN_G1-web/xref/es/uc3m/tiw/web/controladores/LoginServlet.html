19  @WebServlet("/LoginServlet")
20  public class LoginServlet extends HttpServlet {
21  	private static final long serialVersionUID = 1L;
22      private static final String ENTRADA_JSP = "/Admin_Administrador.jsp";
23      private static final String LOGIN_JSP = "/Admin_Login.jsp";
24      
25      private Usuario usuario;
26      private ArrayList<Usuario> usuarios;
27      
28      public void init() throws ServletException {
29      	usuario = new Usuario (2, "Miguel", "Solera", 1, "miguel@uc3m.es", "565543324", "VISA", 0, "1234");
30      	usuarios = new ArrayList<Usuario>();
31      	usuarios.add(usuario);
32      }
33      	/**
34  	 * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
35  	 */
36  	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
37  			// TODO Auto-generated method stub
38  			String salir = request.getParameter("accion");
39  				if (salir != null && !salir.equals("")){
40  					request.getSession().invalidate();
41  				}
42  			this.getServletContext().getRequestDispatcher(LOGIN_JSP).forward(request, response);
43  			
44  	}
45  
46  	/**
47  	 * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
48  	 */
49  	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
50  		// TODO Auto-generated method stub
51  		String user = request.getParameter("usuariologin");
52  		String password = request.getParameter("passwordlogin");
53  		String pagina = "";
54  		String mensaje = "";
55  		pagina = LOGIN_JSP;
56  		
57  		HttpSession sesion = request.getSession();
58  		ServletContext context = sesion.getServletContext();
59  		Usuario u = comprobarUsuario(user, password);
60  		
61  		if (u != null){
62  			pagina = ENTRADA_JSP;
63  			sesion.setAttribute("usuarios", usuarios);
64  			sesion.setAttribute("usuario", u);
65  			context.setAttribute("acceso", "ok");
66  		}
67  		
68  		else {
69  			mensaje = "Usuario o password incorrectos";
70  			request.setAttribute("mensaje", mensaje);
71  		}
72  		
73  		this.getServletContext().getRequestDispatcher(pagina).forward(request, response);
74  		
75  	}
76  	private Usuario comprobarUsuario(String user, String password) {
77  		// TODO Auto-generated method stub
78  		Usuario u = null;
79  		for (Usuario usuario : usuarios){
80  			if (user.equals(usuario.getEmail()) && password.equals(usuario.getClave())){
81  				u = usuario;
82  				break;
83  			}
84  		}
85  		return u;
86  	}
